# technically this is overwriting program memory
loada 0
loadi ra, 10
store ra 0
loadi ra, 11
store ra 1
loadi ra, 23
store ra 2
loadi ra, 8
store ra 3
loadi ra, 3
store ra 4

loadi bp 4

.add_loop:
load.bp ra, 0
add ra, 1
store.bp ra, 0
sub bp, 1
mov ra, bp
jmp.igz.r .add_loop

loadi bp 4
xor rb, rb

.sum_loop:
mov ra, rb
load.bp rb, 0
add ra, rb
sub bp, 1
mov rb, ra
mov ra, bp
jmp.igz.r .sum_loop

out rb
halt
