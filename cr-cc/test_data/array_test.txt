# create array var
.static 2 var
.static 1 second

loada .var

# load three numbers
loadi ra, 10
store ra, .var[0]
loadi ra, 11
store ra, .var[1]
loadi ra, 22
store ra, .second


# sum up single number
loadi ra, 0
load rb, .second
add rb, ra
# keep sum in rb

# for loop to add up array
loadi bp, 1
.array_loop:
load.bp ra, .var
add rb, ra
sub bp, 1
mov ra, bp
jmp.r.igz .array_loop

out rb
halt
