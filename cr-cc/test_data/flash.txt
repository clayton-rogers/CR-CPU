loada 0

### all on ###
.top:
loadi ra, 0xff
out ra


### loop for 400 count ###
loadi ra, 0x90  #0x190 == 400
loadi.h ra, 0x01
.top_loop:
sub ra, 1
jmp.r.igz .top_loop


### all off ###
loadi ra, 0x00
out ra


### loop for another 400 count ###
loadi ra, 0x90  #0x190 == 400
loadi.h ra, 0x01
.top_loop2:
sub ra, 1
jmp.igz.r .top_loop2

### start the whole thing again
jmp .top
